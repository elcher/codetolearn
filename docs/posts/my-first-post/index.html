class Solution:
    def removeElement(self, nums: List[int], val: int) -> int:
        nums.sort()
        left, right = 0, len(nums) - 1
        count = 0
        while left <= right:
            pivot = left + (right - left) // 2
            if nums[pivot] == val:
                count += 1
                while pivot + 1 < len(nums):
                    nums[pivot] = nums[pivot + 1]
                    pivot += 1
                right -= 1
            elif nums[pivot] < val:
                left = pivot + 1
            else:
                right = pivot - 1
        return len(nums) - count
